#include <stdio.h>
#include <pthread.h>
#include <sys/time.h>
#include <semaphore.h>
#include <stdlib.h>
#include <unistd.h>
#include <time.h>
#include <string.h>


/*program readersandwriters*/
int readcount;
double time_main;
double time_in, time_out;
long pid;
semaphore x = 1, wsem = 1;

void reader(){
    while(1){
        semWait(x);
        readcount++;
        if(readcount == 1)
            semWait(wsem);
        semSignal(x);
        readdata();
        printf("pid%ld r %d %d", pid, time_in, time_out);
        semWait(x);
        readcount--;
        if(readcount == 0)
            semSignal(wsem);
        semSignal(x);
    }
}

void writer(){
    while(1){
        semWait(wsem);
        writedata();
        printf("pid%ld w %d %d", pid, time_in, time_out);
        semSignal(wsem);
    }
}

void readdata(){
    time_in = gettimeofday() - time_main;
    pid = pthread_self();
    time_out = gettimeofday() - time_main;
}

void writedata(){
    time_in = gettimeofday() - time_main;
    pid = pthread_self();
    time_out = gettimeofday() - time_main;
}

void main(){
    time_main = gettimeofday();
    readcount = 0;
    parbegin(reader, writer);
}
